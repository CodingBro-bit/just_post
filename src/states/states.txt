 const [posts , setPosts] = useState(mytodos)
const handleCreate = () => {

    const newpost = {

    };
    setPosts([...posts , newpost]);
  }

  const handleDelete = (postId) => {
      const updated = posts.filter(item => item.id !== postId);
      setPosts(updated);
  }

  const handleIncrease = (postId) => {
    
    const find = posts.map( item => {
      if(item.id === postId){
        return {
          ...item , likes:item.likes+1
        }
      }else{
        return item;
      }
    });
    setPosts(find)
  }

  const handleDecrease = (postId) => {
    const find = posts.map( item => {
      if(item.id === postId){
        return {
          ...item , likes:item.likes-1
        }
      }else{
        return item;
      }
    });
    setPosts(find)
  }
